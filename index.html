สรุปเนื้อหาครบถ้วนของ Metronic Admin Dashboard Theme — ระบบ Design System ที่ใช้ Bootstrap 5 พร้อม Custom Components, SASS Variables, Dark Mode และ Layout System ที่ยืดหยุ่นสูง

Metronic คืออะไร?

Metronic คือ Admin Dashboard Theme ระดับ Premium จาก Keenthemes ที่รองรับหลาย Framework ได้แก่ HTML/Bootstrap, Vue.js, React, Angular, ASP.NET Core, Laravel, Django, Flask และอื่นๆ

ฟีเจอร์เด่น:

  • ใช้ Bootstrap 5 เป็น Base พร้อม Custom Extensions มากมาย
  • ระบบ SASS Variables กว่า 1,200+ ตัวแปรที่ปรับแต่งได้
  • รองรับ Dark Mode ด้วย CSS Variables และ data-bs-theme
  • มากกว่า 16 Dashboard Layouts สำหรับหลากหลาย Use Cases
  • ระบบ Component ที่ครบถ้วน: Cards, Tables, Forms, Charts, Widgets
  • โครงสร้าง Layout ที่ยืดหยุ่น: Sidebar, Header, Toolbar, Content

// ===========================

// Danger colors

$danger-active:   #D81A48 !default;
$danger-light:    #FFEEF3 !default;
$danger-clarity:  rgba(#F8285A, 0.2) !default;

// Info colors

// Theme Color Map
$theme-colors: (
    "light":     $light,

CSS Variables ที่ Generate อัตโนมัติ

// _root.scss - Generate CSS Custom Properties

:root {
    // สร้าง CSS variables จาก SASS maps อัตโนมัติ
    @each $name, $value in $theme-colors {
        --bs-#{$name}: #{$value};
    }

    @each $name, $value in $theme-active-colors {
        --bs-#{$name}-active: #{$value};
    }

    @each $name, $value in $theme-light-colors {
        --bs-#{$name}-light: #{$value};
    }
}

// Extended Spacers (Bootstrap default = 0-5)
$spacer: 1rem !default; // ~14px

$spacers: (
    0:  0,
    1:  ($spacer * .25),    // 3.5px
    2:  ($spacer * .5),     // 7px
    3:  ($spacer * .75),    // 10.5px
    4:  ($spacer * 1),      // 14px
    5:  ($spacer * 1.25),   // 17.5px
    6:  ($spacer * 1.5),    // 21px
    7:  ($spacer * 1.75),   // 24.5px
    8:  ($spacer * 2),      // 28px
    9:  ($spacer * 2.25),   // 31.5px
    10: ($spacer * 2.5),    // 35px
    // ... ต่อไปจนถึง 20
    20: ($spacer * 5),      // 69px
);

// ใช้งาน: p-1, m-5, px-10, mb-20 เป็นต้น

Dark Mode

Metronic ใช้ Bootstrap 5.3 CSS Variables + data-bs-theme สำหรับ Dark Mode

Dark Mode Implementation

<!-- Enable Dark Mode -->
<html data-bs-theme="dark">

<!-- Toggle via JavaScript -->
<script>
// Get current theme
const currentTheme = document.documentElement
    .getAttribute('data-bs-theme');

// Toggle theme
function toggleTheme() {
    const html = document.documentElement;
    const current = html.getAttribute('data-bs-theme');
    const next = current === 'dark' ? 'light' : 'dark';
    html.setAttribute('data-bs-theme', next);
    localStorage.setItem('theme', next);
}

// Load saved theme
const saved = localStorage.getItem('theme');
if (saved) {
    document.documentElement
        .setAttribute('data-bs-theme', saved);
}
</script>
